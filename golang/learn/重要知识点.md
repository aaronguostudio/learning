# go basic types
- bool
- string
- int int8 int16 int32 int64
- unit unit8 unit16 unit32 unit64 uintptr
- byte  // alias for unit8
- rune  // alias for int32, represents a Unicode code point
- float32 fload64

- 0 for numeric types
- false for boolean types
- "" for strings
- nil for pointers
- nil for slices
- nil for maps
